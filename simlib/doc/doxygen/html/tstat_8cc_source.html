64void TStat::operator () (double x)
65{
66 if (Time<tl) SIMLIB_warning(TStatNotInitialized);
67 double tt = xl*(double(Time)-tl);
68 sxt += tt;
69 sx2t += xl*tt;
70 xl = x;
71 tl = Time;
72 if(++n==1) min=max=x; // TODO: check
73 else
74 {
75 if(x<min) min = x;
76 if(x>max) max = x;
77 }
78}
