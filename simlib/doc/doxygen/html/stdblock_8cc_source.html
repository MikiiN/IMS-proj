130double Qntzr::Value()
131{
132 AlgLoopDetector _(this);
133 double x = InputValue();
134 double zn = (x > 0) ? 1 : ((x < 0) ? -1 : 0);
135 double k = zn * floor((fabs(x) - step / 2) / step + 1);

186Hyst::Hyst(Input i, double _p1, double _p2, double _y1, double _y2, double _tga)
187 : Status(i), p1(_p1), p2(_p2), y1(_y1), y2(_y2), tga(_tga)

192void Hyst::Eval()
193{
194 double x = InputValue();
195 double ys = stl;
196 ys = max(ys, y1); // in the case of bad initialization (out of range)
197 ys = min(ys, y2); // can be leaved out if correct init
198 double yn2 = (x - p2) * tga;
199 double yn1 = (x - p1) * tga;
200 double y = ys;
201 if (ys <= yn2)
202 y = min(yn2, y2);
203 if (ys >= yn1)
204 y = max(yn1, y1);
205 st = y; // new status
206 ValueOK = true;
207}

229void Relay::Eval()
230{
231 double x = InputValue();
232 double y;
233
234 if (x < p1)
235 y = y1;
236 else if (x < p2)
237 y = stl;
238 else if (x < p3)
239 y = 0;
240 else if (x < p4)
241 y = stl;
242 else
243 y = y2;
244 if (y != stl)
245 ContractStep(); // forces step shortening
246 st = y;
247 ValueOK = true;
