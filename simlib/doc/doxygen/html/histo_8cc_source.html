99////////////////////////////////////////////////////////////////////////////
100// operator () - value recording
101//
102void Histogram::operator () (double x)
103{
104 stat(x);
105 if(x<low)
106 {
107 dptr[0]++;
108 return;
109 }
110 unsigned ix = unsigned((x-low)/step);
111 if(ix>count)
112 dptr[count+1]++;
113 else
114 dptr[ix+1]++;
115}

120void Histogram::Init(double l, double s, unsigned c)
121{
122 Dprintf(("Histogram::Init(%g,%g,%i)",l,s,c));
123 low = l; // init low limit
124 if(s<=0) SIMLIB_error(HistoStepError);
125 step = s; // init step
126 if(c==0 || c>MAXHISTOCOUNT) SIMLIB_error(HistoCountError);
127 if (dptr && count!=c)
128 {
129 delete[] dptr;
130 dptr = 0;
131 }
132 if (!dptr)
133 {
134 count = c; // init count
135 dptr = Alloc(count+2);
136 }
137 Clear(); // init
138}
